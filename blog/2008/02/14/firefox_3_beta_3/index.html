Firefox_3_beta_3

 

sudo make && sudo make install

wget -c http://releases.mozilla.org/pub/mozilla.org/firefox/releases/3.0b3/source/firefox-3.0b3-source.tar.bz2

 

tar -xjvf  firefox-3.0b3-source.tar.bz2

sudo apt-get build-dep iceweasel

sudo apt-get install libdbus-glib-1-dev libcurl4-openssl-dev -y

./configure –prefix=/usr/local/firefox-3.0b3 –enable-application=browser

make  && make install

bin: firefox -P ff3 -no-remote
